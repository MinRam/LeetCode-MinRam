class Solution {
public:
    int mySqrt(int x) {
        if (0 == x || 1 == x) return x;
        for(int i = 2 ; i < x; ++i){
            if(i^i > x) return i-1
        }
        return -1;
    }
};

class Solution {
public:
    int mySqrt(int x) {
        if (0 == x || 1 == x) return x;

        double res = 1, pre = 0;
        while (abs(res - pre) > 1e-6) {
            pre = res;
            res = (res + x / res) / 2;
        }
        return int(res);
    }
};
